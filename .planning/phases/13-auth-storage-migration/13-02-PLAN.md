**supabase/functions/migrate-firebase-auth/index.ts:**

This Edge Function runs in Deno. Firebase Admin SDK does NOT work in Deno, so use the Firebase REST API to verify the ID token.

**CRITICAL: The Edge Function MUST return `access_token` and `refresh_token` in its response.** The client calls `supabase.auth.setSession({ access_token, refresh_token })` to establish the Supabase session without requiring the user to re-verify. This is the core of the "silent migration" locked decision.

**Session token generation strategy:**
After creating/finding the Supabase auth user, generate real GoTrue session tokens by:
1. Set a temporary random password on the user via `auth.admin.updateUserById(userId, { password: tempPassword })`
2. Sign in via the GoTrue token endpoint: `POST /auth/v1/token?grant_type=password` with `{ phone, password: tempPassword }`
3. This returns real `access_token` + `refresh_token` from GoTrue
4. Immediately clear the temporary password: `auth.admin.updateUserById(userId, { password: '' })`
5. Return the tokens to the client

**Phone lookup strategy (scalable):**
Do NOT use `auth.admin.listUsers()` which fetches all users and fails at scale. Instead, query the `auth.users` table directly via the service role client:
```typescript
const { data: existingAuthUser } = await supabaseAdmin
  .rpc('get_auth_user_by_phone', { phone_number: phone });
```
Or use a direct Postgres function created in the migration:
```sql
CREATE OR REPLACE FUNCTION get_auth_user_by_phone(phone_number TEXT)
RETURNS TABLE(id UUID, phone TEXT) AS $$
  SELECT id, phone FROM auth.users WHERE phone = phone_number LIMIT 1;
$$ LANGUAGE sql SECURITY DEFINER;
```

**Full implementation:**

```typescript
import { createClient } from 'https://esm.sh/@supabase/supabase-js@2';

const corsHeaders = {
  'Access-Control-Allow-Origin': '*',
  'Access-Control-Allow-Headers': 'authorization, x-client-info, apikey, content-type',
};

Deno.serve(async (req) => {
  if (req.method === 'OPTIONS') {
    return new Response('ok', { headers: corsHeaders });
  }

  try {
    const { firebaseToken } = await req.json();
    if (!firebaseToken) {
      return new Response(JSON.stringify({ error: 'Missing firebaseToken' }), {
        status: 400,
        headers: { ...corsHeaders, 'Content-Type': 'application/json' },
      });
    }

    // 1. Verify Firebase ID token via Google Identity Toolkit REST API
    const FIREBASE_API_KEY = Deno.env.get('FIREBASE_API_KEY')!;
    const verifyResponse = await fetch(
      `https://identitytoolkit.googleapis.com/v1/accounts:lookup?key=${FIREBASE_API_KEY}`,
      {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ idToken: firebaseToken }),
      }
    );

    if (!verifyResponse.ok) {
      return new Response(JSON.stringify({ error: 'Invalid Firebase token' }), {
        status: 401,
        headers: { ...corsHeaders, 'Content-Type': 'application/json' },
      });
    }

    const verifyData = await verifyResponse.json();
    const firebaseUser = verifyData.users?.[0];
    if (!firebaseUser) {
      return new Response(JSON.stringify({ error: 'Firebase user not found' }), {
        status: 404,
        headers: { ...corsHeaders, 'Content-Type': 'application/json' },
      });
    }

    const firebaseUid = firebaseUser.localId;
    const phone = firebaseUser.phoneNumber;

    if (!phone) {
      return new Response(JSON.stringify({ error: 'No phone number on Firebase account' }), {
        status: 400,
        headers: { ...corsHeaders, 'Content-Type': 'application/json' },
      });
    }

    // 2. Create Supabase admin client with service_role key
    const supabaseUrl = Deno.env.get('SUPABASE_URL')!;
    const supabaseServiceKey = Deno.env.get('SUPABASE_SERVICE_ROLE_KEY')!;
    const supabaseAdmin = createClient(supabaseUrl, supabaseServiceKey);

    // 3. Check if user already migrated (lookup by firebase_uid in users table)
    const { data: existingUser } = await supabaseAdmin
      .from('users')
      .select('id')
      .eq('firebase_uid', firebaseUid)
      .single();

    let supabaseUserId: string;
    let isNewMigration = false;

    if (existingUser) {
      supabaseUserId = existingUser.id;
    } else {
      // First migration -- check if phone already exists in Supabase Auth
      // Use direct query instead of listUsers() for scalability
      const { data: authUserByPhone } = await supabaseAdmin
        .from('users')
        .select('id')
        .eq('phone_number', phone)
        .single();

      if (authUserByPhone) {
        // Phone exists in app users table -- link firebase_uid
        supabaseUserId = authUserByPhone.id;
      } else {
        // Create new Supabase auth user with phone
        const tempPassword = crypto.randomUUID();
        const { data: newAuthUser, error: createError } = await supabaseAdmin.auth.admin.createUser({
          phone,
          phone_confirm: true,
          password: tempPassword,
          user_metadata: { firebase_uid: firebaseUid },
        });

        if (createError) {
          return new Response(JSON.stringify({ error: 'Failed to create Supabase user: ' + createError.message }), {
            status: 500,
            headers: { ...corsHeaders, 'Content-Type': 'application/json' },
          });
        }

        supabaseUserId = newAuthUser.user.id;
        isNewMigration = true;
      }

      // Update users table with firebase_uid mapping
      await supabaseAdmin
        .from('users')
        .update({ firebase_uid: firebaseUid })
        .eq('id', supabaseUserId);
    }

    // 4. Generate real session tokens for the client
    // Strategy: set temp password -> sign in via GoTrue -> clear password -> return tokens
    const tempPassword = crypto.randomUUID();

    // Set temporary password on the user
    const { error: updatePwError } = await supabaseAdmin.auth.admin.updateUserById(supabaseUserId, {
      password: tempPassword,
    });

    if (updatePwError) {
      return new Response(JSON.stringify({ error: 'Failed to generate session' }), {
        status: 500,
        headers: { ...corsHeaders, 'Content-Type': 'application/json' },
      });
    }

    // Sign in with the temporary password to get real GoTrue tokens
    const tokenResponse = await fetch(`${supabaseUrl}/auth/v1/token?grant_type=password`, {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'apikey': supabaseServiceKey,
      },
      body: JSON.stringify({ phone, password: tempPassword }),
    });

    if (!tokenResponse.ok) {
      // Clear temp password even on failure
      await supabaseAdmin.auth.admin.updateUserById(supabaseUserId, { password: '' });
      return new Response(JSON.stringify({ error: 'Failed to generate session tokens' }), {
        status: 500,
        headers: { ...corsHeaders, 'Content-Type': 'application/json' },
      });
    }

    const tokenData = await tokenResponse.json();

    // Immediately clear the temporary password (security)
    await supabaseAdmin.auth.admin.updateUserById(supabaseUserId, { password: '' });

    // 5. Return session tokens to client
    return new Response(JSON.stringify({
      success: true,
      migrated: isNewMigration || !existingUser,
      supabaseUserId,
      access_token: tokenData.access_token,
      refresh_token: tokenData.refresh_token,
      expires_in: tokenData.expires_in,
    }), {
      status: 200,
      headers: { ...corsHeaders, 'Content-Type': 'application/json' },
    });

  } catch (error) {
    return new Response(JSON.stringify({ error: error.message }), {
      status: 500,
      headers: { ...corsHeaders, 'Content-Type': 'application/json' },
    });
  }
});
```
