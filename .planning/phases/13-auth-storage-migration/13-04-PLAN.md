**scripts/migrate-firebase-storage.ts:**

This script runs locally via `npx ts-node scripts/migrate-firebase-storage.ts` (or `npx tsx`). It requires:
- `firebase-admin` (for listing/downloading files from Firebase Storage)
- `@supabase/supabase-js` (for uploading to Supabase Storage)
- Both are available in the project already

**Environment variables needed (read from .env or command line):**
- `FIREBASE_SERVICE_ACCOUNT_PATH` -- path to Firebase service account JSON file
- `FIREBASE_STORAGE_BUCKET` -- Firebase Storage bucket name (e.g., `re-lapse-fa89b.appspot.com` or `flick-prod-49615.appspot.com`)
- `SUPABASE_URL` -- Supabase project URL
- `SUPABASE_SERVICE_ROLE_KEY` -- Service role key for admin access

**Script structure:**

```typescript
import * as admin from 'firebase-admin';
import { createClient } from '@supabase/supabase-js';
import * as fs from 'fs';
import * as path from 'path';

// --- Config ---
const BATCH_SIZE = 50;        // Files to process in parallel
const PROGRESS_FILE = 'scripts/.migration-progress.json';
const BUCKETS_MAP = {
  'photos': { supabaseBucket: 'photos', prefix: 'photos/' },
  'profile-photos': { supabaseBucket: 'profile-photos', prefix: 'profile-photos/' },
  'selects': { supabaseBucket: 'selects', prefix: 'selects/' },
  'comment-images': { supabaseBucket: 'comment-images', prefix: 'comment-images/' },
};

// --- Initialize clients ---
// Firebase Admin with service account
// Supabase with service_role key

// --- Progress tracking ---
// Load progress from PROGRESS_FILE if exists (resume capability)
// Track: { completed: string[], failed: { path: string, error: string }[], total: number }

// --- Main migration flow ---
async function migrate() {
  console.log('Starting Firebase -> Supabase Storage migration...');

  for (const [firebasePrefix, config] of Object.entries(BUCKETS_MAP)) {
    console.log(`\nMigrating ${firebasePrefix}...`);

    // 1. List all files in Firebase Storage under prefix
    const [files] = await admin.storage().bucket().getFiles({ prefix: firebasePrefix });
    console.log(`Found ${files.length} files in ${firebasePrefix}`);

    // 2. Filter out already-migrated files (from progress file)
    const remaining = files.filter(f => !progress.completed.includes(f.name));
    console.log(`${remaining.length} files remaining`);

    // 3. Process in batches of BATCH_SIZE
    for (let i = 0; i < remaining.length; i += BATCH_SIZE) {
      const batch = remaining.slice(i, i + BATCH_SIZE);

      await Promise.all(batch.map(async (file) => {
        try {
          // Download from Firebase
          const [buffer] = await file.download();

          // Determine Supabase path
          const supabasePath = file.name.startsWith(config.prefix)
            ? file.name.substring(config.prefix.length)
            : file.name;

          // Detect content type from extension
          const ext = path.extname(file.name).toLowerCase();
          const contentTypeMap: Record<string, string> = {
            '.jpg': 'image/jpeg', '.jpeg': 'image/jpeg', '.png': 'image/png',
            '.webp': 'image/webp', '.mov': 'video/quicktime', '.mp4': 'video/mp4',
          };
          const contentType = contentTypeMap[ext] || 'application/octet-stream';

          // Upload to Supabase
          const { error } = await supabase.storage
            .from(config.supabaseBucket)
            .upload(supabasePath, buffer, {
              contentType,
              cacheControl: 'public, max-age=31536000',
              upsert: true,
            });

          if (error) throw error;
          progress.completed.push(file.name);
        } catch (err) {
          progress.failed.push({ path: file.name, error: err.message });
          console.error(`Failed: ${file.name} - ${err.message}`);
        }
      }));

      // Save progress after each batch
      fs.writeFileSync(PROGRESS_FILE, JSON.stringify(progress, null, 2));
      console.log(`Progress: ${progress.completed.length}/${files.length} (${progress.failed.length} failed)`);
    }
  }

  // Final report
  console.log('\n=== Migration Complete ===');
  console.log(`Total completed: ${progress.completed.length}`);
  console.log(`Total failed: ${progress.failed.length}`);
  if (progress.failed.length > 0) {
    console.log('Failed files:', progress.failed.map(f => f.path).join('\n'));
  }
}

migrate().catch(console.error);
```

**Key implementation details:**
- Resume capability: progress saved to `scripts/.migration-progress.json` after each batch.
- Parallel batches of 50 files for speed.
- Content type detection from file extension.
- `upsert: true` allows safe re-runs.
- Files are NOT re-compressed to WebP (existing JPEG files stay as JPEG).
- Progress file is gitignored.

**supabase/migrations/20260323000010_update_photo_urls.sql:**

This migration updates all stored Firebase Storage URLs to Supabase Storage CDN URLs. It should be run AFTER the storage migration script (Task 1) has completed.

```sql
-- ================================================================
-- URL Migration: Firebase Storage -> Supabase Storage
-- Run AFTER scripts/migrate-firebase-storage.ts has completed
-- ================================================================

CREATE OR REPLACE FUNCTION migrate_firebase_url(firebase_url TEXT, supabase_project_url TEXT)
RETURNS TEXT AS $$
DECLARE
  decoded_path TEXT;
  bucket TEXT;
  file_path TEXT;
BEGIN
  IF firebase_url IS NULL OR firebase_url = '' THEN
    RETURN firebase_url;
  END IF;

  IF firebase_url LIKE '%supabase%' THEN
    RETURN firebase_url;
  END IF;

  decoded_path := split_part(split_part(firebase_url, '/o/', 2), '?', 1);
  decoded_path := replace(decoded_path, '%2F', '/');

  IF decoded_path LIKE 'photos/%' THEN
    bucket := 'photos';
    file_path := decoded_path;
  ELSIF decoded_path LIKE 'profile-photos/%' THEN
    bucket := 'profile-photos';
    file_path := substring(decoded_path from length('profile-photos/') + 1);
  ELSIF decoded_path LIKE 'selects/%' THEN
    bucket := 'selects';
    file_path := substring(decoded_path from length('selects/') + 1);
  ELSIF decoded_path LIKE 'comment-images/%' THEN
    bucket := 'comment-images';
    file_path := substring(decoded_path from length('comment-images/') + 1);
  ELSE
    RETURN firebase_url;
  END IF;

  RETURN supabase_project_url || '/storage/v1/object/public/' || bucket || '/' || file_path;
END;
$$ LANGUAGE plpgsql IMMUTABLE;

-- NOTE: Replace 'https://YOUR_PROJECT.supabase.co' with your actual project URL before running

UPDATE photos
SET photo_url = migrate_firebase_url(photo_url, 'https://YOUR_PROJECT.supabase.co')
WHERE photo_url LIKE '%firebasestorage.googleapis.com%';

UPDATE users
SET photo_url = migrate_firebase_url(photo_url, 'https://YOUR_PROJECT.supabase.co')
WHERE photo_url LIKE '%firebasestorage.googleapis.com%';

DO $$
BEGIN
  IF NOT EXISTS (
    SELECT 1 FROM information_schema.columns
    WHERE table_name = 'photos' AND column_name = 'storage_path'
  ) THEN
    ALTER TABLE photos ADD COLUMN storage_path TEXT;
  END IF;
END $$;

UPDATE photos
SET storage_path = split_part(photo_url, '/public/photos/', 2)
WHERE photo_url LIKE '%/public/photos/%'
  AND (storage_path IS NULL OR storage_path = '');

CREATE INDEX IF NOT EXISTS idx_photos_storage_path ON photos (storage_path) WHERE storage_path IS NOT NULL;
```
