  <action>
In `App.js`, fix the foreground/background notification handler (around lines 337-354) that currently passes `thumbnailUri: ''` to `startPinnedSnapActivity`.

1. Add `import * as FileSystem from 'expo-file-system';` to the imports section (expo-file-system is already installed in the project).

2. Replace the existing pinned snap handler block. The new logic should:
   a. Extract `pinnedThumbnailUrl` from `notifData` (this is the Firebase Storage URL sent by the Cloud Function)
   b. If `pinnedThumbnailUrl` is present and non-empty, download it to a local temp file using `FileSystem.downloadAsync()` with a destination path of `${FileSystem.cacheDirectory}pinned-thumb-${notifData.pinnedActivityId}.jpg`
   c. Pass the downloaded file's `uri` as `thumbnailUri` to `startPinnedSnapActivity`
   d. If download fails or no URL is available, still call `startPinnedSnapActivity` with `thumbnailUri: ''` (graceful fallback -- widget shows placeholder, which is better than no Live Activity at all)

The updated block should look like:
```javascript
if (Platform.OS === 'ios' && notifData?.pinned === 'true' && notifData?.pinnedActivityId) {
  (async () => {
    try {
      // Download thumbnail from Firebase Storage URL to local cache
      let thumbnailUri = '';
      if (notifData.pinnedThumbnailUrl) {
        try {
          const localPath = `${FileSystem.cacheDirectory}pinned-thumb-${notifData.pinnedActivityId}.jpg`;
          const downloadResult = await FileSystem.downloadAsync(
            notifData.pinnedThumbnailUrl,
            localPath
          );
          thumbnailUri = downloadResult.uri;
          logger.info('App: Downloaded pinned snap thumbnail', { uri: thumbnailUri });
        } catch (dlErr) {
          logger.warn('App: Failed to download pinned thumbnail, proceeding without', {
            error: dlErr.message,
          });
        }
      }

      const laResult = await startPinnedSnapActivity({
        activityId: notifData.pinnedActivityId,
        senderName: notifData.senderName || 'Someone',
        caption: notifData.caption || null,
        conversationId: notifData.conversationId,
        thumbnailUri,
      });
      logger.info('App: Live Activity start result', laResult);
    } catch (err) {
      logger.error('App: Failed to start Live Activity', { error: err.message });
    }
  })();
}
```

IMPORTANT: The `startPinnedSnapActivity` import from `liveActivityService` is already present at the top of App.js. Do NOT remove it. Only add the `FileSystem` import.
