**Step 4: Conversion pattern for each trigger type.**

For onCreate (v1) ‚Üí onDocumentCreated (v2):
```javascript
// BEFORE:
exports.myFunc = functions.firestore.document('col/{docId}').onCreate(async (snap, context) => {
  const data = snap.data();
  const docId = context.params.docId;
  // ...
});

// AFTER:
exports.myFunc = onDocumentCreated('col/{docId}', async (event) => {
  if (!event.data) return;
  const data = event.data.data();
  const docId = event.params.docId;
  // ...
});
```

For onWrite (v1) ‚Üí onDocumentWritten (v2):
```javascript
// BEFORE:
exports.myFunc = functions.firestore.document('col/{docId}').onWrite(async (change, context) => {
  const before = change.before.data();
  const after = change.after.data();
  const docId = context.params.docId;
  // ...
});

// AFTER:
exports.myFunc = onDocumentWritten('col/{docId}', async (event) => {
  const before = event.data.before.data();
  const after = event.data.after.data();
  const docId = event.params.docId;
  // ...
});
```

For pubsub.schedule (v1) ‚Üí onSchedule (v2):
```javascript
// BEFORE:
exports.myFunc = functions.pubsub.schedule('every 2 minutes').onRun(async (context) => {
  // ...
});

// AFTER:
exports.myFunc = onSchedule({ schedule: 'every 2 minutes' }, async (event) => {
  // ...
});
```
