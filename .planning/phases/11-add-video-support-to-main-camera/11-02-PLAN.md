   ```javascript
   import { createVideoPlayer } from 'expo-video';
   import { manipulateAsync, SaveFormat } from 'expo-image-manipulator';

   const generateVideoThumbnail = async (videoUri) => {
     try {
       // Step 1: Create a transient player from the local file URI
       const player = createVideoPlayer(videoUri);
       // Step 2: Generate a single thumbnail at time 0 (first frame)
       const thumbnails = await player.generateThumbnailsAsync([0]);
       // Step 3: Release the player immediately — we only needed the thumbnail
       player.release();
       if (!thumbnails || thumbnails.length === 0) return null;
       const thumbUri = thumbnails[0].uri; // local file URI of the extracted frame
       // Step 4: Resize to tiny placeholder (20px wide) and convert to base64 JPEG
       const manipulated = await manipulateAsync(
         thumbUri,
         [{ resize: { width: 20 } }],
         { compress: 0.5, format: SaveFormat.JPEG, base64: true }
       );
       return `data:image/jpeg;base64,${manipulated.base64}`;
     } catch (error) {
       logger.warn('generateVideoThumbnail failed, continuing without thumbnail', { error: error.message });
       return null; // Graceful fallback — video will still load without a placeholder
     }
   };
   ```
   NOTE: `createVideoPlayer` (from expo-video) creates a standalone player instance that can operate on a local file URI without a React component. This is the non-hook equivalent of `useVideoPlayer` suitable for use outside React components. If `createVideoPlayer` is not exported (check expo-video API), fall back to using `VideoThumbnails.getThumbnailAsync` as a last resort, or return null with a warning log.

   ```javascript
   const uploadQueueItem = async (item) => {
     const { id, userId, mediaType = 'photo' } = item;

     const extension = isVideo

     // Generate thumbnail — use existing generateThumbnail for photos, new generateVideoThumbnail for videos

       : await generateThumbnail(mediaUri);

     };
     await setDoc(photoRef, docData);
   ```

```javascript
/**
 * VideoMuteContext - Global mute state for video playback
 *
 * Tracks whether the user has unmuted videos. When a user taps to unmute
 * a video in the feed, the unmuted state persists as they navigate to
 * PhotoDetail, stories, and back to the feed. Resets to muted on app
 * background or when the user explicitly mutes.
 *
 * Usage:
 * - Wrap app root with <VideoMuteProvider>
 * - Components call useVideoMute() to get { isMuted, toggleMute, setMuted }
 */
import React, { createContext, useContext, useState, useCallback } from 'react';

const VideoMuteContext = createContext({
  isMuted: true,
  toggleMute: () => {},
  setMuted: () => {},
});

export const useVideoMute = () => useContext(VideoMuteContext);

export const VideoMuteProvider = ({ children }) => {
  const [isMuted, setIsMuted] = useState(true); // Start muted (feed autoplay muted)

  const toggleMute = useCallback(() => {
    setIsMuted(prev => !prev);
  }, []);

  const setMuted = useCallback((value) => {
    setIsMuted(value);
  }, []);

  return (
    <VideoMuteContext.Provider value={{ isMuted, toggleMute, setMuted }}>
      {children}
    </VideoMuteContext.Provider>
  );
};

export default VideoMuteContext;
```
