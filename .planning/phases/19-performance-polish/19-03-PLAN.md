TanStack Query optimistic update pattern:
```typescript
useMutation({
  mutationFn: ...,
  onMutate: async (variables) => {
    await queryClient.cancelQueries({ queryKey });
    const previous = queryClient.getQueryData(queryKey);
    queryClient.setQueryData(queryKey, updater);
    return { previous };
  },
  onError: (_err, _vars, context) => {
    queryClient.setQueryData(queryKey, context?.previous);
    Toast.show({ type: 'error', text1: 'Failed message' });
  },
  onSettled: () => {
    queryClient.invalidateQueries({ queryKey });
  },
});
```

      mutationFn: MutationFunction<TData, TVariables>;
      queryKey: QueryKey | ((variables: TVariables) => QueryKey);
      updater: (old: TQueryData | undefined, variables: TVariables) => TQueryData;
      errorMessage: string;
      invalidateKeys?: QueryKey[];
      // Callback passthrough (addresses review MEDIUM concern)
      onSuccess?: (data: TData, variables: TVariables) => void;
      onMutate?: (variables: TVariables) => void;
      onSettled?: (data: TData | undefined, error: Error | null, variables: TVariables) => void;
    }

    // Multi-key updater (for operations like reactions that update multiple caches)
    // Addresses review HIGH concern: reactions need to update photo detail AND feed list
    interface MultiKeyUpdater<TVariables> {
      queryKey: QueryKey | ((variables: TVariables) => QueryKey);
      updater: (old: any, variables: TVariables) => any;
    }

    interface MultiKeyOptions<TData, TVariables> {
      mutationFn: MutationFunction<TData, TVariables>;
      updaters: MultiKeyUpdater<TVariables>[];  // Array of {queryKey, updater} pairs
      errorMessage: string;
      invalidateKeys?: QueryKey[];
      onSuccess?: (data: TData, variables: TVariables) => void;
      onMutate?: (variables: TVariables) => void;
      onSettled?: (data: TData | undefined, error: Error | null, variables: TVariables) => void;
    }
